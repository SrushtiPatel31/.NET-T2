namespace Calc
{
    /// <summary>
    /// Interaction logic for MainWindow.xaml
    /// </summary>
    public partial class MainWindow : Window
    {
        Double f_n, s_n, r_n;
        String seleted_operator;
        public MainWindow()
        {
            InitializeComponent();
        }


        private void num_Click(object sender, RoutedEventArgs e)
        {
            Button ?btn = sender as Button;
            if (tb.Text == "0" && btn.Content.ToString() == ".")
            {
                tb.Text = "0.";
                return;
            }
            if(tb.Text=="0" && btn.Content.ToString() != ".")
            {
                tb.Text = btn.Content.ToString();
                return;
            }
            if(tb.Text.Contains(".") && btn.Content.ToString() == "."){
                return;
            }
            else
            {
                tb.Text += btn.Content.ToString();
            }
        }

        private void equal_Click(object sender, RoutedEventArgs e)
        {
            Double s_n = Convert.ToDouble(tb.Text);
            switch (seleted_operator)
            {
                case "+":
                    r_n = f_n + s_n;
                    tb.Text = r_n.ToString();
                    break;
                case "-":
                    r_n = f_n - s_n;
                    tb.Text = r_n.ToString();
                    break;
                case "*":
                    r_n = f_n * s_n;
                    tb.Text = r_n.ToString();
                    break;
                case "/":
                    r_n = f_n / s_n;
                    tb.Text = r_n.ToString();
                    break;
            }
        }

        private void op_btn_Click(object sender, RoutedEventArgs e)
        {
            Button ?op = sender as Button;
            f_n = Convert.ToDouble(tb.Text);
            seleted_operator = op.Content.ToString();
            tb.Text = "0";
            MessageBox.Show("First  no is " + f_n + "Seleted Operator : " + seleted_operator);
        }

        private void Per_Click(object sender, RoutedEventArgs e)
        {
            tb.Text = (Convert.ToDouble(tb.Text) / 100).ToString();
        }

        private void Neg_Click(object sender, RoutedEventArgs e)
        {
            tb.Text = (Convert.ToDouble(tb.Text)*-1).ToString();
        }

        private void AC_Click(object sender, RoutedEventArgs e)
        {
            tb.Text = "0";
        }
    }
}
